Function: getIntVar

int getIntVar(char charVar)
{
    for (int i = 0; i < varMapSize; i++)
    {
        if (varMap[i].charVar == charVar)
        {
            return varMap[i].intVar;
        }
    }

    // Create new mapping
    varMap[varMapSize].charVar = charVar;
