Function: freeDIMACS

// Free DIMACS formula
void freeDIMACS(DIMACSFormula *formula)
{
    if (formula == NULL)
        return;

    for (int i = 0; i < formula->numClauses; i++)
    {
        free(formula->clauses[i].literals);
    }
    free(formula->clauses);
    free(formula);
}

Frees the dynamically allocated array of literals for each clause.
